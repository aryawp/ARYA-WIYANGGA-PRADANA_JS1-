/ Contoh program JavaScript dengan switch case untuk menentukan hari dalam seminggu berdasarkan nomor hari (0 untuk Minggu, 1 untuk Senin, dst.) let nomorHari = 3; let namaHari; switch (nomorHari) { case 0: namaHari = "Minggu"; break; case 1: namaHari = "Senin"; break; case 2: namaHari = "Selasa"; break; case 3: namaHari = "Rabu"; break; case 4: namaHari = "Kamis"; break; case 5: namaHari = "Jumat"; break; case 6: namaHari = "Sabtu"; break; default: namaHari = "Hari tidak valid"; }
